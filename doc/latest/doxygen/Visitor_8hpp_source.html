661template<class T>
662void Visitor::dispatch(const T& obj) {
663 auto hash = reinterpret_cast<size_t>(&obj);
664 if (visited_.find(hash) != std::end(visited_)) {
665 // Already visited
666 return;
667 }
668
669 visited_.insert(hash);
670 visit(obj);
671}
