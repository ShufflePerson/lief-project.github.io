718template<class T>
719void Visitor::dispatch(const T& obj) {
720 auto hash = reinterpret_cast<size_t>(&obj);
721 if (visited_.find(hash) != std::end(visited_)) {
722 // Already visited
723 return;
724 }
725
726 visited_.insert(hash);
727 visit(obj);
728}
